力扣第一题暴力破解算法，hash待更新

#define _CRT_SECURE_NO_WARNINGS //解决新版VS scanf警告问题
#include <stdio.h>
#include <malloc.h>
#include <stdlib.h>
//暴力破解算法
int* twosum(int* nums, int numsSize, int target, int* returnSize)
{
for (int i = 0; i < numsSize; i++) //俩for走天下
{
for (int j = i + 1; j < numsSize; j++)
{
if (target == nums[i] + nums[j])
{
int* ret = malloc(sizeof(int) * 2);
ret[0] = i;
ret[1] = j;
*returnSize = 2;
return ret;
}
}
}
*returnSize = 0;
return NULL;
}

int main()
{
int* nums = NULL, * ret;
int returnSize;

}
